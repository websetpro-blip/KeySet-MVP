==== ЭТАП 2: НАСТРОИТЬ РОУТИНГ (1 ЧАС) ====

4. Создай frontend/src/App.jsx:

import React, { lazy, Suspense } from 'react';
import { BrowserRouter, Routes, Route, Link } from 'react-router-dom';

// Lazy imports
const AccountsTab = lazy(() => import('./modules/accounts/AccountsTab'));
const MasksTab = lazy(() => import('./modules/masks/MasksTab'));
const ProxyTab = lazy(() => import('./modules/proxy/ProxyTab'));
const DataTab = lazy(() => import('./modules/data/DataTab'));

export default function App() {
  return (
    <BrowserRouter>
      <div className="app">
        <nav>
          <Link to="/accounts">Аккаунты</Link>
          <Link to="/masks">Маски</Link>
          <Link to="/proxy">Прокси</Link>
          <Link to="/data">Данные</Link>
        </nav>
        
        <Suspense fallback={<div>Загрузка...</div>}>
          <Routes>
            <Route path="/accounts" element={<AccountsTab />} />
            <Route path="/masks" element={<MasksTab />} />
            <Route path="/proxy" element={<ProxyTab />} />
            <Route path="/data" element={<DataTab />} />
            <Route path="/" element={<AccountsTab />} />
          </Routes>
        </Suspense>
      </div>
    </BrowserRouter>
  );
}

==== ЭТАП 4: LOCALAGENT (FASTAPI) (2 ЧАСА) ====

6. Создай backend/main.py:

from fastapi import FastAPI
from fastapi.staticfiles import StaticFiles
from fastapi.responses import FileResponse
import uvicorn

app = FastAPI()

# Мок-эндпоинты
@app.get("/api/accounts")
def get_accounts():
    return [
        {"id": 1, "login": "account1@mail.ru", "status": "active"},
        {"id": 2, "login": "account2@mail.ru", "status": "inactive"}
    ]

@app.post("/api/parsing/start")
def start_parsing(phrases: list):
    return {"job_id": 1, "status": "started"}

@app.get("/api/proxies")
def get_proxies():
    return [
        {"id": 1, "ip": "192.168.1.1", "port": 8080},
        {"id": 2, "ip": "192.168.1.2", "port": 8081}
    ]

# Раздача React UI
app.mount("/", StaticFiles(directory="../frontend/dist", html=True), name="static")

if __name__ == "__main__":
    uvicorn.run(app, host="127.0.0.1", port=8765)

==== ЭТАП 5: LAUNCHER (PYWEBVIEW) (1 ЧАС) ====

8. Создай launcher.py (в корне):

import webview
import uvicorn
from threading import Thread

def start_api():
    uvicorn.run("backend.main:app", host="127.0.0.1", port=8765)

if __name__ == '__main__':
    # Запускаем FastAPI в фоне
    Thread(target=start_api, daemon=True).start()
    
    # Открываем локальный UI
    webview.create_window('KeySet', 'http://127.0.0.1:8765')
    webview.start()

a = Analysis(['launcher.py'], pathex=[], binaries=[],
